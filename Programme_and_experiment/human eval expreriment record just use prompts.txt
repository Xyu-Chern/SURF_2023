from typing import List

def separate_paren_groups(paren_string: str) -> List[str]:
    stack = []
    result = []
    current = ""
    for char in paren_string:
        if char == "(":
            stack.append(char)
        elif char == ")":
            stack.pop()
        current += char
        if not stack:
            result.append(current.replace(" ", ""))
            current = ""
    return result

        level = 0

            if c == '(':
                stack.append(c)
                level = max(level, len(stack))

def longest(strings: List[str]) -> Optional[str]:
    if not strings:
        return None
    longest_str = strings[0]
    for string in strings:
        if len(string) > len(longest_str):

    return longest_str

        diff = numbers[i+1] - numbers[i]

from typing import List

def factorize(n: int) -> List[int]:
    factors = []
    d = 2
    while d * d <= n:
        while (n % d) == 0:
            factors.append(d)
            n //= d
        d += 1
    if n > 1:
        factors.append(n)
    return factors

def flip_case(string: str) -> str:

    for char in string:
        if char.isupper():
            new_string += char.lower()
        else:
            new_string += char.upper()

def encode_cyclic(s: str):
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    groups = [(group[1:] + group[0]) if len(group) == 3 else group for group in groups]
    return "".join(groups)

def decode_cyclic(s: str):
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    groups = [(group[-1] + group[:-1]) if len(group) == 3 else group for group in groups]
    return "".join(groups)

def prime_fib(n: int) -> int:
    def is_fib(m: int) -> bool:
        a, b = 0, 1
        while a < m:
            a, b = b, a + b
        return a == m
    count = 0
    i = 2
    while count < n:
        if is_fib(i) and is_prime(i):
            count += 1
        i += 1
    return i - 1

def fib4(n: int):
    a, b, c, d = 0, 0, 2, 0
    if n == 0:
        return a
    elif n == 1:
        return b
    elif n == 2:
        return c
    else:
        for i in range(3, n+1):
            d = a + b + c + d
            a, b, c = b, c, d
        return d

def median(l: list):
    l.sort()
    n = len(l)
    mid = n // 2
    if n % 2 == 0:
        return (l[mid - 1] + l[mid]) / 2
    else:
        return l[mid]

def minSubArraySum(nums):
    curr_sum = min_sum = nums[0]
    for i in range(1, len(nums)):
        curr_sum = min(nums[i], curr_sum + nums[i])
        min_sum = min(min_sum, curr_sum)
    return min_sum

def minPath(grid, k):
    n = len(grid)
    visited = set()
    heap = [(grid[i][0], (i, 0), [grid[i][0]]) for i in range(n)]
    heapq.heapify(heap)
    while heap:
        val, (i, j), path = heapq.heappop(heap)
        if len(path) == k:
            return path
        visited.add((i, j))
        for ni, nj in [(i+1, j), (i-1, j), (i, j+1), (i, j-1)]:
            if 0 <= ni < n and 0 <= nj < n and (ni, nj) not in visited:
                heapq.heappush(heap, (grid[ni][nj], (ni, nj), path + [grid[ni][nj]]))

def is_nested(string):
    stack = []
    for char in string:
        if char == "[":
            stack.append(char)
        elif char == "]":
            if len(stack) > 0 and stack[-1] == "[":
                stack.pop()
            else:
                stack.append(char)
    return len(stack) > 0 and "[" in stack

def int_to_mini_roman(number):
    roman_numerals = {
        1000: 'm',
        900: 'cm',
        500: 'd',
        400: 'cd',
        100: 'c',
        90: 'xc',
        50: 'l',
        40: 'xl',
        10: 'x',
        9: 'ix',
        5: 'v',
        4: 'iv',
        1: 'i'
    }
    result = ''
    for value, numeral in roman_numerals.items():
        while number >= value:
            result += numeral
            number -= value
    return result.lower()
